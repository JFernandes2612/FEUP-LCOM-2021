60  green_blob->y += delta_y;
61  green_blob->direction = (green_blob->direction + 1) % 4;

65  green_blob->action = GREEN_BLOB_NO_ACT;

69  {
70  green_blob->direction = (green_blob->direction + 1) % 4;
71  green_blob->action = GREEN_BLOB_NO_ACT;
72  break;
73  }
74  green_blob->action = GREEN_BLOB_ATTACK;
